# start the agent
eval "$(ssh-agent -s)"

# List keys added to the agent:
ssh-add -l

# If it says “The agent has no identities”, you need to add your key:
ssh-add ~/.ssh/you_privatekey

# Test SSH connection to GITHUB
ssh -T git@github.com
# Hi username! You've successfully authenticated, but GitHub does not provide shell access.

# Clone git repo by SSH:
git clone git@github.com:NIOZ-QingZ/3D_models_WaddenSea.git


# Commit changes
## Check which files have changed:
git status

## 2. Stage the changes:
### a single file
git add filename

### Stage all changes:
git add .

## 3. Commit the changes
git commit -m "add your description" # the "-m" lets you write a short commit message

## 4. Push to repository
git push

### If it is the first push of a new branch:
git push -u origin branch_name

## 5. Pull the newest changes from the remote
git pull
